820template <class T>
821size_t DEREncodeUnsigned(BufferedTransformation &out, T w, byte asnTag = INTEGER)
822{
823 byte buf[sizeof(w)+1];
824 unsigned int bc;
825 if (asnTag == BOOLEAN)
826 {
827 buf[sizeof(w)] = w ? 0xff : 0;
828 bc = 1;
829 }
830 else
831 {
832 buf[0] = 0;
833 for (unsigned int i=0; i<sizeof(w); i++)
834 buf[i+1] = byte(w >> (sizeof(w)-1-i)*8);
835 bc = sizeof(w);
836 while (bc > 1 && buf[sizeof(w)+1-bc] == 0)
837 --bc;
838 if (buf[sizeof(w)+1-bc] & 0x80)
839 ++bc;
840 }
841 out.Put(asnTag);
842 size_t lengthBytes = DERLengthEncode(out, bc);
843 out.Put(buf+sizeof(w)+1-bc, bc);
844 return 1+lengthBytes+bc;
845}

856template <class T>
857void BERDecodeUnsigned(BufferedTransformation &in, T &w, byte asnTag = INTEGER,
858 T minValue = 0, T maxValue = T(0xffffffff))
859{
860 byte b;
861 if (!in.Get(b) || b != asnTag)
862 BERDecodeError();
863
864 size_t bc;
865 bool definite = BERLengthDecode(in, bc);
866 if (!definite)
867 BERDecodeError();
868 if (bc > in.MaxRetrievable()) // Issue 346
869 BERDecodeError();
870 if (asnTag == BOOLEAN && bc != 1) // X.690, 8.2.1
871 BERDecodeError();
872 if ((asnTag == INTEGER || asnTag == ENUMERATED) && bc == 0) // X.690, 8.3.1 and 8.4
873 BERDecodeError();
874
875 SecByteBlock buf(bc);
876
877 if (bc != in.Get(buf, bc))
878 BERDecodeError();
879
880 // This consumes leading 0 octets. According to X.690, 8.3.2, it could be non-conforming behavior.
881 // X.690, 8.3.2 says "the bits of the first octet and bit 8 of the second octet ... (a) shall
882 // not all be ones and (b) shall not all be zeros ... These rules ensure that an integer value
883 // is always encoded in the smallest possible number of octet".
884 // We invented AER (Alternate Encoding Rules), which is more relaxed than BER, CER, and DER.
885 const byte *ptr = buf;
886 while (bc > sizeof(w) && *ptr == 0)
887 {
888 bc--;
889 ptr++;
890 }
891 if (bc > sizeof(w))
892 BERDecodeError();
893
894 w = 0;
895 for (unsigned int i=0; i<bc; i++)
896 w = (w << 8) | ptr[i];
897
898 if (w < minValue || w > maxValue)
899 BERDecodeError();
900}
