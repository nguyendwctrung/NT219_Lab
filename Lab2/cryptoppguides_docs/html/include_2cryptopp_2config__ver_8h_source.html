55// Compiler version macros
56
57// Apple and LLVM Clang versions. Apple Clang version 7.0 roughly equals
58// LLVM Clang version 3.7. Also see https://gist.github.com/yamaya/2924292
59#if defined(__clang__) && defined(__apple_build_version__)
60# define CRYPTOPP_APPLE_CLANG_VERSION (__clang_major__ * 10000 + __clang_minor__ * 100 + __clang_patchlevel__)
61#elif defined(__clang__) && defined(_MSC_VER)
62# define CRYPTOPP_MSVC_CLANG_VERSION (__clang_major__ * 10000 + __clang_minor__ * 100 + __clang_patchlevel__)
63#elif defined(__clang__)
64# define CRYPTOPP_LLVM_CLANG_VERSION (__clang_major__ * 10000 + __clang_minor__ * 100 + __clang_patchlevel__)
65#endif
66
67// Clang pretends to be other compilers. The compiler gets into
68// code paths that it cannot compile. Unset Clang to save the grief.
69// Also see http://github.com/weidai11/cryptopp/issues/147.
70
71#if defined(__GNUC__) && !defined(__clang__)
72# undef CRYPTOPP_APPLE_CLANG_VERSION
73# undef CRYPTOPP_LLVM_CLANG_VERSION
74# define CRYPTOPP_GCC_VERSION (__GNUC__ * 10000 + __GNUC_MINOR__ * 100 + __GNUC_PATCHLEVEL__)
75#endif
76
77#if defined(__xlc__) || defined(__xlC__) && !defined(__clang__)
78# undef CRYPTOPP_LLVM_CLANG_VERSION
79# define CRYPTOPP_XLC_VERSION ((__xlC__ / 256) * 10000 + (__xlC__ % 256) * 100)
80#endif
81
82#if defined(__INTEL_COMPILER) && !defined(__clang__)
83# undef CRYPTOPP_LLVM_CLANG_VERSION
84# define CRYPTOPP_INTEL_VERSION (__INTEL_COMPILER)
85#endif
86
87#if defined(_MSC_VER) && !defined(__clang__)
88# undef CRYPTOPP_LLVM_CLANG_VERSION
89# define CRYPTOPP_MSC_VERSION (_MSC_VER)
90#endif
