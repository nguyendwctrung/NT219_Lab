149inline word ShiftWordsLeftByBits(word *r, size_t n, unsigned int shiftBits)
150{
151 CRYPTOPP_ASSERT (shiftBits<WORD_BITS);
152 word u, carry=0;
153 if (shiftBits)
154 for (size_t i=0; i<n; i++)
155 {
156 u = r[i];
157 r[i] = (u << shiftBits) | carry;
158 carry = u >> (WORD_BITS-shiftBits);
159 }
